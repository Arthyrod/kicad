#include <18F4550.h>
#fuses HS, NOWDT, NOPROTECT, NOLVP, NOVREGEN, NOPBADEN, NOMCLR
#use delay(clock=4000000)
int8 codigos_anodo_comum[16] = {
   0xC0,  // 0 ? 1100 0000 ? acende a,b,c,d,e,f
   0xF9,  // 1 ? 1111 1001 ? acende b,c
   0xA4,  // 2 ? 1010 0100 ? acende a,b,d,e,g
   0xB0,  // 3 ? 1011 0000 ? acende a,b,c,d,g
   0x99,  // 4 ? 1001 1001 ? acende b,c,f,g
   0x92,  // 5 ? 1001 0010 ? acende a,c,d,f,g
   0x82,  // 6 ? 1000 0010 ? acende a,c,d,e,f,g
   0xF8,  // 7 ? 1111 1000 ? acende a,b,c
   0x80,  // 8 ? 1000 0000 ? acende todos
   0x90,  // 9 ? 1001 0000 ? acende a,b,c,d,f,g
   0x88,  // A ? 1000 1000 ? acende a,b,c,e,f,g
   0x83,  // B ? 1000 0011 ? acende c,d,e,f,g
   0xC6,  // C ? 1100 0110 ? acende a,d,e,f
   0xA1,  // D ? 1010 0001 ? acende b,c,d,e,g
   0x86,  // E ? 1000 0110 ? acende a,d,e,f,g
   0x8E   // F ? 1000 1110 ? acende a,e,f,g
};
int calcular_codigo(int8 valor_codigo){
   int8 codigo_display;
   switch(valor_codigo){
         case 0x0:
            codigo_display = codigos_anodo_comum[0];
            break;
         case 0x1:
            codigo_display = codigos_anodo_comum[1];
            break;
        case 0x2:
            codigo_display = codigos_anodo_comum[2];
            break;
        case 0x3:
            codigo_display = codigos_anodo_comum[3];
            break;
        case 0x4:
            codigo_display = codigos_anodo_comum[4];
            break;
        case 0x5:
            codigo_display = codigos_anodo_comum[5];
            break;
        case 0x6:
            codigo_display = codigos_anodo_comum[6];
            break;
        case 0x7:
            codigo_display = codigos_anodo_comum[7];
            break;
        case 0x8:
            codigo_display = codigos_anodo_comum[8];
            break;
        case 0x9:
            codigo_display = codigos_anodo_comum[9];
            break;
        case 0xA:
            codigo_display = codigos_anodo_comum[10];
            break;
        case 0xB:
            codigo_display = codigos_anodo_comum[11];
            break;
        case 0xC:
            codigo_display = codigos_anodo_comum[12];
            break;
        case 0xD:
            codigo_display = codigos_anodo_comum[13];
            break;
        case 0xE:
            codigo_display = codigos_anodo_comum[14];
            break;
        case 0xF:
            codigo_display = codigos_anodo_comum[15];
            break;
      }   
    return codigo_display;  
   }
void main() {
   setup_adc_ports(NO_ANALOGS);
   setup_adc(ADC_OFF);
   int8 num_1;
   int num_2;
   int8 display1;
   int8 display2;
   set_tris_b(0x00);   // PORTB saída (LED)
   set_tris_a(0x00);
   set_tris_d(0xFF);   // PORTD entrada (botão)
   output_a(0x00);
   output_b(0xFF); // LED apagado ao iniciar
   while(TRUE) {
      int8 leitura = input_d();
      display1 = leitura & 0x0F; 
      display2 = (leitura>>4) & 0x0F;
      num_1 = calcular_codigo(display1);
      num_2 = calcular_codigo(display2);
      output_b(num_1);
      output_a(0x01);
      delay_ms(1);
      output_b(num_2);
      output_a(0x02);
      delay_ms(1);
   }
}
